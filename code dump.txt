function animateCounters() {

    if (text.match(/^\d+/)) {
      const num = parseInt(text);
      const suffix = text.replace(/[\d,]+/, '');
      let current = 0;
      const step = Math.max(1, Math.floor(num / 40));
      const interval = setInterval(() => {
        current += step;
        if (current >= num) {
          current = num;
          clearInterval(interval);
        }
        el.textContent = current + suffix;
      }, 30);
    }
  });

  });
