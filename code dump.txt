    <div class="card cur">
      <h3>◣ What current state costs</h3>
      <p><strong>Correlated blind spots.</strong> Recon and NAV read the same stale price, so both go green while the output is wrong. Two controls agree because they share a source.</p>
      <p><strong>Cause-first triage.</strong> The accountant works every feed error by hand, ranked by noise, not by whether the NAV actually moved. Effort scales with alerts, not with risk.</p>
      <p><strong>No hard stop.</strong> A wrong NAV can publish because nothing structurally prevents release — the Knight Capital failure mode: a missing halt.</p>
    </div>
    <div class="card fut">
      <h3>◢ What future state changes</h3>
      <p><strong>Uncorrelated redundancy.</strong> An independent shadow-NAV path catches what recon can't, because it doesn't share the blind spot. Defense in depth, by different authors.</p>
      <p><strong>Symptom-first, "too good" alerting.</strong> The feed surfaces outputs that look wrong — including a return that's implausibly <em>smooth</em> — ranked by materiality, not error count.</p>
      <p><strong>The strike-gate holds.</strong> NAV cannot publish while a material exception is open. The human signs the release; the model never posts, reverses, or signs.</p>

<script>
const W=1180,H=560;
const svg=d3.select("#canvas");

// ---------- data ----------
const current={
  title:"Current state — recon queue, cause-first",
  badge:"Effort scales with alerts",
  badgeCls:"cur",
  legend:[
    {c:"var(--green)",t:"green control / passes"},
    {c:"var(--amber)",t:"manual review"},
    {c:"var(--red)",t:"silent blind spot"}
  ],
  nodes:[
    {id:"feed",x:90,y:280,label:"Price + expense feeds",sub:"vendor, custody, accruals",col:"var(--txt-dim)"},
    {id:"nav",x:340,y:180,label:"Accounting platform",sub:"strikes NAV off last snapshot",col:"var(--txt-dim)"},
    {id:"recon",x:340,y:380,label:"Recon controls",sub:"three-way, balanced",col:"var(--green)"},
    {id:"queue",x:610,y:280,label:"Recon break queue",sub:"every break, ranked by noise",col:"var(--amber)"},
    {id:"fa",x:860,y:280,label:"Fund accountant",sub:"works each cause by hand",col:"var(--amber)"},
    {id:"pub",x:1075,y:280,label:"NAV published",sub:"no hard stop",col:"var(--txt-dim)"}
  ],
  links:[
    {s:"feed",t:"nav",kind:"normal"},
    {s:"feed",t:"recon",kind:"normal"},
    {s:"nav",t:"queue",kind:"normal"},
    {s:"recon",t:"queue",kind:"normal"},
    {s:"queue",t:"fa",kind:"manual"},
    {s:"fa",t:"pub",kind:"manual"},
    {s:"feed",t:"nav",kind:"blind"} // the silent stale path
  ],
  blindNote:{x:215,y:130,t:"stale price flows through — both NAV and recon read it, both stay green"}
};

const future={
  title:"Future state — strike-gate, symptom-first",
  badge:"Effort scales with risk",
  badgeCls:"fut",
  legend:[
    {c:"var(--green)",t:"deterministic proof"},
    {c:"var(--cyan)",t:"uncorrelated check"},
    {c:"var(--gold)",t:"human sign-off"},
    {c:"var(--red)",t:"gate held"}
  ],
  nodes:[
    {id:"feed",x:80,y:280,label:"Price + expense feeds",sub:"vendor, custody, accruals",col:"var(--txt-dim)"},
    {id:"recompute",x:300,y:150,label:"Deterministic recompute",sub:"reproduces NAV to the cent",col:"var(--green)"},
    {id:"shadow",x:300,y:290,label:"Shadow-NAV (independent path)",sub:"different price snapshot",col:"var(--cyan)"},
    {id:"iso",x:300,y:430,label:"Isolation Forest + EWMA",sub:"multivariate outliers",col:"var(--cyan)"},
    {id:"symptom",x:560,y:290,label:"Symptom feed",sub:'"too smooth" / divergence, ranked',col:"var(--cyan)"},
    {id:"gate",x:790,y:290,label:"Strike-gate",sub:"holds while exception open",col:"var(--red)"},
    {id:"fa",x:1000,y:160,label:"Fund accountant",sub:"resolves 6, not 40 — signs release",col:"var(--gold)"},
    {id:"pub",x:1000,y:430,label:"NAV published",sub:"proof + signature attached",col:"var(--green)"}
  ],
  links:[
    {s:"feed",t:"recompute",kind:"normal"},
    {s:"feed",t:"shadow",kind:"check"},
    {s:"feed",t:"iso",kind:"check"},
    {s:"recompute",t:"symptom",kind:"normal"},
    {s:"shadow",t:"symptom",kind:"check"},
    {s:"iso",t:"symptom",kind:"check"},
    {s:"symptom",t:"gate",kind:"normal"},
    {s:"gate",t:"fa",kind:"human"},
    {s:"fa",t:"gate",kind:"return"},
    {s:"gate",t:"pub",kind:"release"}
  ],
  blindNote:null
};

const palette={
  normal:"#2c4358",
  manual:"var(--amber)",
  blind:"var(--red)",
  check:"var(--cyan)",
  human:"var(--gold)",
  return:"var(--gold)",
  release:"var(--green)"
};

function nodeById(data,id){return data.nodes.find(n=>n.id===id);}

let particleTimers=[];
function clearTimers(){particleTimers.forEach(t=>clearInterval(t));particleTimers=[];}

function render(data){
  clearTimers();
  svg.selectAll("*").remove();

  // defs: glow
  const defs=svg.append("defs");
  ["var(--cyan)","var(--gold)","var(--red)","var(--green)"].forEach((c,i)=>{
    const f=defs.append("filter").attr("id","glow"+i).attr("x","-50%").attr("y","-50%").attr("width","200%").attr("height","200%");
    f.append("feGaussianBlur").attr("stdDeviation","3.2").attr("result","b");
    const m=f.append("feMerge");m.append("feMergeNode").attr("in","b");m.append("feMergeNode").attr("in","SourceGraphic");
  });

  // links
  const linkLayer=svg.append("g");
  const pathFor=(l)=>{
    const a=nodeById(data,l.s),b=nodeById(data,l.t);
    const mx=(a.x+b.x)/2;
    if(l.kind==="return"){ // curve back, offset
      return `M${b.x-14},${b.y+18} C${mx},${b.y+70} ${mx},${a.y+70} ${a.x+14},${a.y+18}`;
    }
    return `M${a.x},${a.y} C${mx},${a.y} ${mx},${b.y} ${b.x},${b.y}`;
  };

  data.links.forEach((l,i)=>{
    const col=palette[l.kind]||"#2c4358";
    const dash=(l.kind==="blind")?"2,6":(l.kind==="check"?"1,7":null);
    const p=linkLayer.append("path")
      .attr("d",pathFor(l))
      .attr("fill","none")
      .attr("stroke",col)
      .attr("stroke-width",l.kind==="blind"?2:1.6)
      .attr("opacity",0);
    if(dash)p.attr("stroke-dasharray",dash);
    const len=p.node().getTotalLength();
    p.attr("stroke-dasharray",dash?dash:len+" "+len)
     .attr("stroke-dashoffset",dash?0:len)
     .transition().delay(200+i*120).duration(700)
     .attr("opacity",l.kind==="blind"?0.85:0.55)
     .attr("stroke-dashoffset",0)
     .on("end",function(){if(dash)d3.select(this).attr("stroke-dasharray",dash);});

    // flowing particles (skip blind = static danger line)
    if(l.kind!=="blind"){
      const t=setInterval(()=>{
        const dot=linkLayer.append("circle").attr("r",2.6).attr("fill",col).attr("opacity",0.9);
        dot.transition().duration(1600).ease(d3.easeLinear)
          .attrTween("transform",()=>{return tt=>{const pt=p.node().getPointAtLength(tt*len);return `translate(${pt.x},${pt.y})`;};})
          .attr("opacity",0).remove();
      },900+Math.random()*400);
      particleTimers.push(t);
    }
  });

  // blind-spot annotation
  if(data.blindNote){
    const g=svg.append("g").attr("opacity",0);
    g.append("text").attr("x",data.blindNote.x).attr("y",data.blindNote.y)
      .attr("class","axis-note").attr("fill","var(--red)").attr("text-anchor","middle")
      .text("◬ silent stale path");
    g.append("text").attr("x",data.blindNote.x).attr("y",data.blindNote.y+15)
      .attr("class","axis-note").attr("fill","#8a5560").attr("text-anchor","middle")
      .attr("font-size","9px").text("recon & NAV share the blind spot");
    g.transition().delay(1400).duration(800).attr("opacity",1);
  }

  // nodes
  const nodeLayer=svg.append("g");
  data.nodes.forEach((n,i)=>{
    const g=nodeLayer.append("g").attr("transform",`translate(${n.x},${n.y})`).attr("opacity",0);
    const w=Math.max(150, n.label.length*8.2);
    const isGate=n.id==="gate";
    g.append("rect").attr("x",-w/2).attr("y",-26).attr("width",w).attr("height",52)
      .attr("rx",9).attr("fill","var(--panel2)")
      .attr("stroke",n.col).attr("stroke-width",isGate?1.8:1.1)
      .attr("filter",isGate?"url(#glow2)":null);
    if(isGate){
      g.append("rect").attr("x",-w/2-3).attr("y",-29).attr("width",w+6).attr("height",58)
        .attr("rx",11).attr("fill","none").attr("stroke","var(--red)").attr("stroke-width",0.8).attr("opacity",0.35)
        .attr("stroke-dasharray","4,4");
    }
    g.append("circle").attr("cx",-w/2+13).attr("cy",-9).attr("r",3.5).attr("fill",n.col);
    g.append("text").attr("x",-w/2+24).attr("y",-5).attr("class","node-label").text(n.label);
    g.append("text").attr("x",-w/2+24).attr("y",13).attr("class","node-sub").text(n.sub);
    g.transition().delay(300+i*130).duration(600).attr("opacity",1);
  });

  // stage caption strip
  svg.append("text").attr("x",W/2).attr("y",H-14).attr("text-anchor","middle")
     .attr("class","axis-note")
     .text(data===current ? "feeds → platform & recon → break queue → accountant → publish   ·   no structural stop"
                          : "feeds → 3 uncorrelated checks → symptom feed → STRIKE-GATE → human sign-off → publish with proof");
}

const states={cur:current,fut:future};
function show(k){
  const d=states[k];
  document.getElementById("bt").textContent=d.title;
  const bd=document.getElementById("bd");bd.textContent=d.badge;bd.className="badge "+d.badgeCls;
  document.getElementById("btn-cur").classList.toggle("active",k==="cur");
  document.getElementById("btn-fut").classList.toggle("active",k==="fut");
  const leg=document.getElementById("leg");leg.innerHTML="";
  d.legend.forEach(L=>{
    const s=document.createElement("span");
    s.innerHTML=`<span class="dot" style="background:${L.c}"></span>${L.t}`;
    leg.appendChild(s);
  });
  render(d);
}
show('cur');
</script>
