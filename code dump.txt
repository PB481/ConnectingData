/* ============================================================
   ENGINE  —  generic. Renders schema, handles branching,
   suggestion-then-confirm, progress, and JSON export.
============================================================ */
const answers = {};     // id -> value (string | array)
const confirmed = {};   // id -> true once user confirms
const isSuggested = {}; // id -> true if current value came from a suggestion (unconfirmed)
const form = document.getElementById("form");

function visibleQuestions(){ return SCHEMA.filter(q=>!q.show || q.show(answers)); }

function applySuggestions(){
  visibleQuestions().forEach(q=>{
    if(!q.suggest) return;
    if(confirmed[q.id]) return;               // never overwrite confirmed
    if(answers[q.id]!==undefined && !isSuggested[q.id]) return; // user typed their own
    const s = q.suggest(answers);
    if(s===null||s===undefined) return;
    answers[q.id] = s;
    isSuggested[q.id] = true;
  });
}

function setValue(id,val){
  answers[id]=val;
  isSuggested[id]=false;     // user touched it -> no longer a raw suggestion
  confirmed[id]=false;       // editing un-confirms
  applySuggestions();
  render();
}

function confirm(id){ confirmed[id]=true; isSuggested[id]=false; render(); }

function render(){
  const vis = visibleQuestions();
  form.innerHTML="";
  let prevAnswered = true;

  vis.forEach((q,i)=>{
    const hasVal = answers[q.id]!==undefined && answers[q.id]!=="" &&
                   !(Array.isArray(answers[q.id]) && answers[q.id].length===0);
    const needsConfirm = q.type!=="text" && q.type!=="number" && q.type!=="textarea"
                          ? hasVal : hasVal;   // all answered fields require a confirm tick
    const gated = !prevAnswered;

    const card=document.createElement("div");
    card.className="card"+(gated?" gated":"");
    card.innerHTML=`<span class="qnum">Q${i+1}</span>`;

    const qt=document.createElement("div");qt.className="qtext";qt.textContent=q.q;card.appendChild(qt);
    if(q.help){const h=document.createElement("div");h.className="qhelp";h.textContent=q.help;card.appendChild(h);}

    const flagged = isSuggested[q.id];

    if(q.type==="single"||q.type==="multi"){
      const wrap=document.createElement("div");wrap.className="opts";
      q.options.forEach(o=>{
        const sel = q.type==="single" ? answers[q.id]===o.value
                                      : Array.isArray(answers[q.id])&&answers[q.id].includes(o.value);
        const opt=document.createElement("label");
        opt.className="opt"+(sel?(flagged?" suggested":" sel"):"");
        opt.innerHTML=`<input type="${q.type==="single"?"radio":"checkbox"}" name="${q.id}" ${sel?"checked":""}>
          <span>${o.label}</span>${sel&&flagged?'<span class="tag">suggested</span>':''}`;
        opt.querySelector("input").addEventListener("change",()=>{
          if(q.type==="single") setValue(q.id,o.value);
          else{
            const cur=Array.isArray(answers[q.id])?[...answers[q.id]]:[];
            const ix=cur.indexOf(o.value);
            ix>-1?cur.splice(ix,1):cur.push(o.value);
            setValue(q.id,cur);
          }
        });
        wrap.appendChild(opt);
      });
      card.appendChild(wrap);
    } else {
      const lbl=document.createElement("label");lbl.className="flbl";
      lbl.textContent=q.unit?("Value ("+q.unit+")"):"Your answer";
      card.appendChild(lbl);
      let el;
      if(q.type==="textarea"){el=document.createElement("textarea");}
      else{el=document.createElement("input");el.type=q.type==="number"?"number":"text";}
      el.placeholder=q.placeholder||"";
      el.value=answers[q.id]??"";
      if(flagged)el.classList.add("suggested");
      el.addEventListener("input",e=>{
        answers[q.id]= q.type==="number" ? (e.target.value===""?"":Number(e.target.value)) : e.target.value;
        isSuggested[q.id]=false;confirmed[q.id]=false;
        updateMeta();exportBtnState();
      });
      el.addEventListener("blur",()=>{applySuggestions();render();});
      card.appendChild(el);
    }

    if(hasVal){
      const row=document.createElement("div");row.className="confirmrow";
      const note=document.createElement("span");
      note.className="note"+(confirmed[q.id]?" done":"");
      note.textContent = confirmed[q.id] ? "✓ Confirmed" :
        (flagged ? "Suggested from earlier answers — confirm to record" : "Confirm to record");
      const btn=document.createElement("button");
      btn.className="cbtn"+(confirmed[q.id]?" locked":"");
      btn.textContent=confirmed[q.id]?"Confirmed":"Confirm";
      if(!confirmed[q.id])btn.addEventListener("click",()=>confirm(q.id));
      row.appendChild(btn);row.appendChild(note);card.appendChild(row);
    }

    form.appendChild(card);
    prevAnswered = hasVal && confirmed[q.id];
  });

  updateMeta();exportBtnState();
}

function updateMeta(){
  const vis=visibleQuestions();
  const done=vis.filter(q=>confirmed[q.id]).length;
  const pct=vis.length?Math.round(done/vis.length*100):0;
  document.getElementById("pctText").textContent=pct+"%";
  document.getElementById("barFill").style.right=(100-pct)+"%";
  document.getElementById("confText").textContent=done+" of "+vis.length+" confirmed";
  document.getElementById("footStat").textContent=
    done===vis.length&&vis.length?"All sections confirmed — ready to export":"";
}

function exportBtnState(){
  const vis=visibleQuestions();
  const all=vis.length>0 && vis.every(q=>confirmed[q.id]);
  document.getElementById("exportBtn").disabled=!all;
  document.getElementById("encExportBtn").disabled=!all;
}

/* ---- export payload ---- */
function buildPayload(){
  const vis=visibleQuestions();
  const labelFor=(q,v)=>{
    if(q.options){const o=q.options.find(o=>o.value===v);return o?o.label:v;}
    return v;
  };
  const fields={};
  vis.forEach(q=>{
    let val=answers[q.id], label;
    if(Array.isArray(val)) label=val.map(v=>labelFor(q,v));
    else label=labelFor(q,val);
    fields[q.id]={question:q.q,value:val,label,confirmed:!!confirmed[q.id]};
  });
  return {
    schema_version:"1.0",
    form:"fund_launch_onboarding",
    generated_at:new Date().toISOString(),
    record_complete:vis.every(q=>confirmed[q.id]),
    answers:Object.fromEntries(vis.map(q=>[q.id,answers[q.id]])), // flat, DB-friendly
    fields                                                         // rich, with labels + audit
  };
}

  Object.keys(confirmed).forEach(k=>delete confirmed[k]);
  Object.keys(isSuggested).forEach(k=>delete isSuggested[k]);
  applySuggestions();render();
};

/* ============================================================
   ENCRYPTION  —  AES-256-GCM via Web Crypto. No libraries.
   Envelope format (itself JSON, safe to email):
   { fmt:"fundq-enc-v1", kdf:"PBKDF2-SHA256", iter, salt, iv, ct }
   All binary fields base64-encoded.
============================================================ */
const PBKDF2_ITER = 250000;
const enc = new TextEncoder();
const dec = new TextDecoder();
const b64 = buf => btoa(String.fromCharCode(...new Uint8Array(buf)));
const unb64 = s => Uint8Array.from(atob(s), c=>c.charCodeAt(0));

async function deriveKey(pass, salt, iter){
  const base = await crypto.subtle.importKey("raw", enc.encode(pass), "PBKDF2", false, ["deriveKey"]);
  return crypto.subtle.deriveKey(
    {name:"PBKDF2", salt, iterations:iter, hash:"SHA-256"},
    base, {name:"AES-GCM", length:256}, false, ["encrypt","decrypt"]);
}

async function encryptPayload(obj, pass){
  const salt = crypto.getRandomValues(new Uint8Array(16));
  const iv   = crypto.getRandomValues(new Uint8Array(12));
  const key  = await deriveKey(pass, salt, PBKDF2_ITER);
  const ct   = await crypto.subtle.encrypt({name:"AES-GCM", iv}, key, enc.encode(JSON.stringify(obj)));
  return {fmt:"fundq-enc-v1", kdf:"PBKDF2-SHA256", iter:PBKDF2_ITER,
          salt:b64(salt), iv:b64(iv), ct:b64(ct), generated_at:new Date().toISOString()};
}

async function decryptPayload(envelope, pass){
  if(envelope.fmt!=="fundq-enc-v1") throw new Error("Not a recognised encrypted file.");
  const key = await deriveKey(pass, unb64(envelope.salt), envelope.iter||PBKDF2_ITER);
  const pt  = await crypto.subtle.decrypt({name:"AES-GCM", iv:unb64(envelope.iv)}, key, unb64(envelope.ct));
  return JSON.parse(dec.decode(pt));
}
