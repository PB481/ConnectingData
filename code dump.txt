const CAT_COLORS = { novel:"#C62828", hybrid:"#EF6C00", routine:"#2E7D32" };

let activeRoute = null;

function nodePos(id) {
  const n = NODES[id]; if (!n) return {x:0,y:0};
  return { x: PAD_L + n.stage*COL_W, y: PAD_T + n.row*ROW_H };
}
function nodeCentre(id) {
  const p = nodePos(id);
  return { x: p.x + NODE_W/2, y: p.y + NODE_H/2 };
}

function routePathD(fromId, toId) {
  const a = nodeCentre(fromId), b = nodeCentre(toId);
  const mx = a.x + (b.x - a.x)*0.55;
  return `M${a.x+NODE_W/2+2},${a.y} L${mx},${a.y} L${mx},${b.y} L${b.x-NODE_W/2-2},${b.y}`;
}

function drawRoutes() {
  const lg = document.getElementById("linesGroup");
  const sg = document.getElementById("stepsGroup");
  lg.innerHTML = ""; sg.innerHTML = "";

  ROUTES.forEach(route => {
    const isActive = activeRoute === route.id;
    const isDimmed = activeRoute && !isActive;

    const rg = document.createElementNS("http://www.w3.org/2000/svg","g");
    rg.style.opacity = isDimmed ? "0.06" : isActive ? "1" : "0.18";
    rg.style.transition = "opacity 0.3s";

    route.path.slice(0,-1).forEach((fromId,i) => {
      const toId = route.path[i+1];
      // Glow
      if (isActive) {
        const glow = document.createElementNS("http://www.w3.org/2000/svg","path");
        glow.setAttribute("d", routePathD(fromId, toId));
        glow.setAttribute("fill","none"); glow.setAttribute("stroke",route.color);
        glow.setAttribute("stroke-width","6"); glow.setAttribute("stroke-linecap","round");
        glow.setAttribute("opacity","0.12");
        rg.appendChild(glow);
      }
      const p = document.createElementNS("http://www.w3.org/2000/svg","path");
      p.setAttribute("d", routePathD(fromId, toId));
      p.setAttribute("fill","none"); p.setAttribute("stroke",route.color);
      p.setAttribute("stroke-width", isActive?"2.5":"2"); p.setAttribute("stroke-linecap","round");
      rg.appendChild(p);

      // Mid dot
      if (isActive) {
        const a = nodeCentre(fromId), b = nodeCentre(toId);
        const dot = document.createElementNS("http://www.w3.org/2000/svg","circle");
        dot.setAttribute("cx",(a.x+b.x)/2+10); dot.setAttribute("cy",(a.y+b.y)/2);
        dot.setAttribute("r","2.5"); dot.setAttribute("fill",route.color);
        rg.appendChild(dot);
      }
    });
    lg.appendChild(rg);

    // Step numbers
    if (isActive) {
      route.path.forEach((nodeId, i) => {
        const c = nodeCentre(nodeId);
        const circ = document.createElementNS("http://www.w3.org/2000/svg","circle");
        circ.setAttribute("cx", c.x - NODE_W/2 - 14); circ.setAttribute("cy", c.y);
        circ.setAttribute("r","10"); circ.setAttribute("fill",route.color);
        sg.appendChild(circ);
        const num = document.createElementNS("http://www.w3.org/2000/svg","text");
        num.setAttribute("x", c.x - NODE_W/2 - 14); num.setAttribute("y", c.y + 3.5);
        num.setAttribute("text-anchor","middle"); num.setAttribute("fill","white");
        num.setAttribute("font-size","10"); num.setAttribute("font-weight","700");
        num.setAttribute("font-family","'JetBrains Mono', monospace");
        num.textContent = i+1;
        sg.appendChild(num);
      });
    }
  });

  // Node opacity
  Object.keys(NODES).forEach(id => {
    const g = document.getElementById("node-"+id);
    const rect = document.getElementById("nodeRect-"+id);
    const sub = document.getElementById("nodeSub-"+id);
    const ts = TYPE_STYLES[NODES[id].type];
    if (activeRoute) {
      const isOn = ROUTES.find(r=>r.id===activeRoute)?.path.includes(id);
      g.style.opacity = isOn ? "1" : "0.25";
      rect.setAttribute("fill", isOn ? ts.fill : "white");
      rect.setAttribute("stroke-width", isOn ? "2" : "1");
      if (sub) sub.style.opacity = isOn ? "1" : "0";
    } else {
      g.style.opacity = "1";
      rect.setAttribute("fill","white");
      rect.setAttribute("stroke-width","1");
      if (sub) sub.style.opacity = "0";
    }
  });
}

function buildRouteBar() {
  const bar = document.getElementById("routeBar");
  bar.innerHTML = "";
  ROUTES.forEach((r,i) => {
    const btn = document.createElement("button");
    btn.className = "route-btn" + (activeRoute === r.id ? " active" : "");
    btn.style.setProperty("--rc", r.color);
    btn.innerHTML = `<div class="rname">${r.name}</div><div class="rcat" style="color:${CAT_COLORS[r.cat]}">${r.cat.toUpperCase()}</div>`;
    btn.onclick = () => { activeRoute = activeRoute === r.id ? null : r.id; render(); };
    bar.appendChild(btn);
  });
}

function buildInfoBar() {
  const el = document.getElementById("infoBar");
  if (!activeRoute) {
    el.innerHTML = '<div class="info-placeholder">Select a use case above to trace its route through the pipeline</div>';
    return;
  }
  const r = ROUTES.find(rt => rt.id === activeRoute);
  const catColor = CAT_COLORS[r.cat];
  const chain = r.path.map((p,i) => {
    const ts = TYPE_STYLES[NODES[p].type];
    const label = NODES[p].label.replace(/\n/g," ");
    return (i>0?'<span style="color:#ccc;margin:0 3px">→</span>':'') +
      `<span class="stop" style="color:${ts.text}">${label}</span>`;
  }).join("");
  const noteHtml = r.note ? `<span class="note"> — ${r.note}</span>` : "";

  el.innerHTML = `
    <div class="info-bar">
      <div class="route-chain">
        <div class="title-row">
          <div class="bar" style="background:${r.color}"></div>
          <h3>${r.name}</h3>
          <span class="cat-badge" style="color:${catColor};background:${catColor}10;border:1px solid ${catColor}25">${r.cat.toUpperCase()}</span>
        </div>
        <div class="chain"><strong style="color:#555">Route (${r.path.length} stops): </strong>${chain}${noteHtml}</div>
      </div>
      <div class="metrics">
        <div class="metric"><div class="mlabel">MANUAL</div><div class="mval" style="color:#C62828">${r.manual}</div></div>
        <div class="metric"><div class="mlabel">SYSTEM</div><div class="mval" style="color:#2E7D32">${r.time}</div></div>
        <div class="metric"><div class="mlabel">COST</div><div class="mval" style="color:#F57C00">${r.cost}</div></div>
      </div>
    </div>`;
}
