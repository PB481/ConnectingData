<script>
let SQL, db, cols=[], rows=[];
const $=id=>document.getElementById(id);
const status=(m,e=false)=>{const s=$('status'); s.textContent=m; s.className='status'+(e?' err':'')};

initSqlJs({locateFile:f=>`https://cdnjs.cloudflare.com/ajax/libs/sql.js/1.10.2/${f}`})
  .then(s=>{SQL=s; status('Ready. Load a CSV to begin.')})
  .catch(e=>status('Failed to load SQLite engine: '+e,true));

// Minimal CSV parser (handles quotes, commas, newlines, escaped quotes)
function parseCSV(text){
  const delim = text.indexOf('\t')>-1 && text.indexOf(',')===-1 ? '\t' : ',';
  const out=[]; let row=[], cur='', q=false;
  for(let i=0;i<text.length;i++){
    const c=text[i], n=text[i+1];
    if(q){
      if(c==='"'&&n==='"'){cur+='"';i++}
      else if(c==='"'){q=false}
      else cur+=c;
    } else {
      if(c==='"')q=true;
      else if(c===delim){row.push(cur);cur=''}
      else if(c==='\n'){row.push(cur);out.push(row);row=[];cur=''}
      else if(c==='\r'){}
      else cur+=c;
    }
  }
  if(cur!==''||row.length){row.push(cur);out.push(row)}
  return out.filter(r=>r.length>1||r[0]!=='');
}

function uniqCols(raw){
  const seen={}; return raw.map((c,i)=>{
    let name=(c||`col${i+1}`).trim().replace(/[^A-Za-z0-9_]/g,'_')||`col${i+1}`;
    if(/^[0-9]/.test(name))name='_'+name;
    if(seen[name]!==undefined){seen[name]++;name+='_'+seen[name]} else seen[name]=0;
    return name;
  });
}

$('csv').addEventListener('change',e=>{
  const f=e.target.files[0]; if(!f)return;
  const r=new FileReader();
  r.onload=()=>{
    try{
      const parsed=parseCSV(r.result);
      if(parsed.length<1){status('CSV appears empty',true);return}
      cols=uniqCols(parsed[0]);
      const body=parsed.slice(1);
      if(db)db.close();
      db=new SQL.Database();
      db.run(`CREATE TABLE data (${cols.map(c=>`"${c}" TEXT`).join(',')})`);
      const stmt=db.prepare(`INSERT INTO data VALUES (${cols.map(()=>'?').join(',')})`);
      body.forEach(row=>{
        const vals=cols.map((_,i)=>row[i]!==undefined?row[i]:null);
        stmt.run(vals);
      });
      stmt.free();
      ['run','addRow','json','dbExport','sql'].forEach(id=>$(id).disabled=false);
      status(`Loaded "${f.name}" → ${body.length} rows × ${cols.length} cols into table "data".`);
      runQuery();
    }catch(err){status('Parse error: '+err.message,true)}
  };
  r.readAsText(f);
});

function runQuery(){
  try{
    const res=db.exec($('sql').value);
    if(!res.length){rows=[];cols=[];render();status('Query OK — no rows returned.');return}
    cols=res[0].columns; rows=res[0].values.map(v=>v.slice());
    render();
    status(`Query returned ${rows.length} row(s).`);
  }catch(err){status('SQL error: '+err.message,true)}
}

function render(){
  const g=$('grid');
  if(!rows.length){g.innerHTML='<div class="empty">No rows</div>';return}
  let h='<table><thead><tr>'+cols.map(c=>`<th>${c}</th>`).join('')+'</tr></thead><tbody>';
  rows.forEach((r,ri)=>{
    h+='<tr>'+r.map((v,ci)=>`<td contenteditable data-r="${ri}" data-c="${ci}">${v??''}</td>`).join('')+'</tr>';
  });
  g.innerHTML=h+'</tbody></table>';
  g.querySelectorAll('td[contenteditable]').forEach(td=>{
    td.addEventListener('blur',()=>{rows[td.dataset.r][td.dataset.c]=td.textContent});
  });
}

$('run').addEventListener('click',runQuery);
$('addRow').addEventListener('click',()=>{rows.push(cols.map(()=>''));render()});

$('json').addEventListener('click',()=>{
  const data=rows.map(r=>Object.fromEntries(cols.map((c,i)=>[c,r[i]])));
  const txt=JSON.stringify(data,null,2);
  $('out').style.display='block'; $('out').textContent=txt;
  dl(new Blob([txt],{type:'application/json'}),'data.json');
  status(`Exported ${data.length} row(s) to JSON.`);
});

$('dbExport').addEventListener('click',()=>{
  dl(new Blob([db.export()],{type:'application/octet-stream'}),'data.sqlite');
  status('Exported SQLite database file.');
});

function dl(blob,name){
  const a=document.createElement('a');
  a.href=URL.createObjectURL(blob); a.download=name; a.click();
  setTimeout(()=>URL.revokeObjectURL(a.href),1000);
}
