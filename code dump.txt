// Super-cluster derivation — 10ish meaningful buckets for layout
// ──────────────────────────────────────────────────────────────────────────
function superCluster(d) {
  if (d.aum_usd_bn >= 2000) return "Trillion-Plus";
  if (d.parent_type === "PE-Alts") return "Alts Specialists";
  if (d.parent_type === "Trust") return "Trust Banks";
  if (d.parent_type === "State-Sponsored") return "State/Pension";
  if (d.parent_type === "Insurance") {
    if (d.region === "Europe" || d.region === "UK") return "Insurance — EU/UK";
    if (d.region === "Japan") return "Insurance — Japan";
    return "Insurance — Other";
  }
  if (d.parent_type === "Bank") {
    if (d.region === "Japan") return "Japan Banks";
    if (d.region === "Europe" || d.region === "UK") return "EU/UK Banks";
    return "Banks — Other";
  }
  if (d.parent_type === "Wealth-Brokerage") return "Wealth-Led";
  if (d.ap_tilt <= 2) return "Passive/Index";
  if (d.ap_tilt >= 4) return "Active Specialists";
  return "Diversified Independent";
}

// Accessor function for the currently selected clustering dimension
function clusterAccessor(mode) {
  if (mode === "super") return d => d.super;
  if (mode === "region") return d => d.region;
  if (mode === "parent_type") return d => d.parent_type;
  if (mode === "asset_class") return d => d.asset_class;
  if (mode === "ta_tilt") return d => "Trad↔Alts tier " + d.ta_tilt;
  if (mode === "ap_tilt") return d => "Active↔Passive tier " + d.ap_tilt;
  return d => d.super;
}

  f.r = r(f.aum_usd_bn);
});

// ──────────────────────────────────────────────────────────────────────────
// Cluster centroid layout — arrange centroids on a deterministic grid,
// sized by member count, sorted by total AUM (biggest cluster top-left)
// ──────────────────────────────────────────────────────────────────────────
function computeCentroids(clusterFn) {
  const groups = d3.rollup(
    FIRMS,
    v => ({ count: v.length, aum: d3.sum(v, d => d.aum_usd_bn) }),
    clusterFn
  );
  const arr = [...groups, ].map(([key, val]) => ({ key, ...val }));
  arr.sort((a, b) => b.aum - a.aum);

  // grid dimensions
  const n = arr.length;
  const cols = Math.ceil(Math.sqrt(n * (WIDTH / HEIGHT)));
  const rows = Math.ceil(n / cols);

  const padX = 80, padY = 70;
  const labelGutter = 26;
  const stepX = (WIDTH - padX * 2) / cols;
  const stepY = (HEIGHT - padY * 2 - labelGutter) / rows;

  const positions = {};
  arr.forEach((c, i) => {
    const col = i % cols;
    const row = Math.floor(i / cols);
    positions[c.key] = {
      x: padX + stepX * col + stepX / 2,
      y: padY + labelGutter + stepY * row + stepY / 2,
      count: c.count,
      aum: c.aum,
      key: c.key,
    };
  });
  return { positions, arr };
}

  else if (currentColor === "parent_type") label = "Parent Type";

  else if (currentColor === "ap_tilt") label = "Active/Passive (1=passive, 5=active)";
  else if (currentColor === "ta_tilt") label = "Trad/Alts (1=traditional, 5=alts)";

  const items = Object.entries(palette).map(([key, col]) =>
    `<div class="legend-item"><span class="legend-swatch" style="background:${col}"></span>${key}</div>`
  ).join("");

  legendEl.innerHTML = `
    <span class="control-label">${label}</span>
    ${items}
    <span style="width: 12px"></span>
    <div class="legend-item"><span class="legend-marker-client"></span>Client</div>
  `;
}

renderLegend();

// ──────────────────────────────────────────────────────────────────────────
// Filtering — search + clients-only
// ──────────────────────────────────────────────────────────────────────────
function applyFilters() {
  const q = currentSearch.trim().toLowerCase();
  nodes.classed("dimmed", d => {
    if (clientsOnly && !CLIENTS.has(d.firm)) return true;
    if (q && !d.firm.toLowerCase().includes(q) && !d.parent.toLowerCase().includes(q)) return true;
    return false;
  });
}

// Control wiring

document.getElementById("ctrl-cluster").addEventListener("change", (e) => {
  currentCluster = e.target.value;
  centroids = computeCentroids(clusterAccessor(currentCluster));
  simulation
    .force("x", d3.forceX(d => centroids.positions[clusterAccessor(currentCluster)(d)].x).strength(0.18))
    .force("y", d3.forceY(d => centroids.positions[clusterAccessor(currentCluster)(d)].y).strength(0.20))
    .alpha(0.7).restart();
  renderClusterLabels();
});

document.getElementById("ctrl-color").addEventListener("change", (e) => {
  currentColor = e.target.value;
  nodes
    .transition().duration(450)
    .attr("fill", d => colorFor(currentColor, d))
    .attr("stroke", d => d3.color(colorFor(currentColor, d)).darker(0.6));
  renderLegend();
});

document.getElementById("ctrl-search").addEventListener("input", (e) => {
  currentSearch = e.target.value;
  applyFilters();
});

document.getElementById("ctrl-clients-only").addEventListener("change", (e) => {
  clientsOnly = e.target.checked;
  applyFilters();
});

document.getElementById("ctrl-reset").addEventListener("click", () => {
  if (CLIENTS.size === 0) return;
  if (!confirm(`Clear all ${CLIENTS.size} tagged clients?`)) return;
  CLIENTS.clear();
  saveClients();
  nodes.classed("client", false);
  updateStats();
  applyFilters();
});
