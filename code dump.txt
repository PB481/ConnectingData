  e.currentTarget.classList.remove('active');
  const dropped = Array.from(e.dataTransfer.files);
  files = files.concat(dropped);
  renderFiles();
  updateContinueBtn();
}

function handleFileSelect(e) {
  files = files.concat(Array.from(e.target.files));
  renderFiles();
  updateContinueBtn();
  e.target.value = '';
}

function removeFile(idx) {
  files.splice(idx, 1);
  renderFiles();
  updateContinueBtn();
}

function renderFiles() {
  const dz = document.getElementById('dropzone');
  const empty = document.getElementById('dropzoneEmpty');
  const chips = document.getElementById('dropzoneFiles');
  if (files.length === 0) {
    empty.classList.remove('hidden');
    chips.classList.add('hidden');
    dz.classList.remove('has-files');
  } else {
    empty.classList.add('hidden');
    chips.classList.remove('hidden');
    dz.classList.add('has-files');
    chips.innerHTML = files.map((f,i) =>
      `<div class="file-chip">📄 ${f.name} <button class="remove" onclick="event.stopPropagation();removeFile(${i})">×</button></div>`
    ).join('') + '<div style="font-size:12px;color:#aaa;font-family:system-ui;padding:8px 0">+ drop more files</div>';
  }
}

function updateContinueBtn() {
  instruction = document.getElementById('instructionInput').value;
  document.getElementById('btnContinue').disabled = !(instruction.trim() || files.length);
}

function quickAction(type, text) {
  document.getElementById('instructionInput').value = text;
  instruction = text;
  requestType = type;
  updateContinueBtn();
  startInterview();
}

// INTERVIEW

function detectType() {
  const txt = (instruction + ' ' + files.map(f=>f.name).join(' ')).toLowerCase();
  if (txt.includes('prospectus')||txt.includes('lpa')||txt.includes('new fund')||txt.includes('launch')||txt.includes('onboard')) return 'fund_launch';
  if (txt.includes('quarterly')||txt.includes('reporting')||txt.includes('investor statement')) return 'quarterly_reporting';
  if (txt.includes('invoice')||txt.includes('bill')) return 'invoice';
  if (txt.includes('fee')||txt.includes('management fee')) return 'fee_calculation';
  if (txt.includes('fair value')||txt.includes('valuation')||txt.includes('pricing instruction')) return 'fair_value';
  return 'general';
}

function startInterview() {
  if (!requestType) requestType = detectType();
  const rt = REQUEST_TYPES[requestType];
  answers = {};
  currentQ = 0;

  document.getElementById('interviewIcon').textContent = rt.icon;
  document.getElementById('interviewTitle').textContent = rt.label;
  document.getElementById('interviewEst').textContent = 'Est. ' + rt.est;

  // Files summary
  const fs = document.getElementById('filesSummary');
  if (files.length) {
    fs.classList.remove('hidden');
    fs.innerHTML = files.map(f => `<span class="file-tag">📄 ${f.name}</span>`).join('');
  } else { fs.classList.add('hidden'); }

  showPhase('phaseInterview');
  renderChat();
}

function renderChat() {
  const qs = AI_QUESTIONS[requestType] || AI_QUESTIONS['general'];
  const chat = document.getElementById('chatArea');
  let html = '';

  // User's original instruction
  if (instruction) {
    html += `<div class="msg-user"><div class="msg-user-bubble">${esc(instruction)}</div></div>`;
  }

  // Q&A pairs
  for (let i = 0; i <= currentQ && i < qs.length; i++) {
    const q = qs[i];
    // AI question
    html += `<div class="msg-ai"><div class="ai-avatar">E</div><div class="msg-ai-bubble">${esc(q.q)}</div></div>`;
    // User answer (if exists)
    if (answers[q.field]) {
      html += `<div class="msg-user"><div class="msg-user-bubble">${esc(answers[q.field])}</div></div>`;
    }
    // Input for current unanswered question
    if (i === currentQ && !answers[q.field]) {
      html += `<div class="answer-row">
        <input class="answer-input" id="answerInput" placeholder="${esc(q.ph)}" onkeydown="if(event.key==='Enter')submitAnswer()" autofocus>
        <button class="btn-send" id="btnSend" onclick="submitAnswer()">→</button>
      </div>`;
    }
  }

  chat.innerHTML = html;

  // Progress dots
  const dots = document.getElementById('progressDots');
  dots.innerHTML = qs.map((_,i) => `<div class="dot${i <= currentQ ? ' filled' : ''}"></div>`).join('');

  // Focus input
  setTimeout(() => {
    const inp = document.getElementById('answerInput');
    if (inp) {
      inp.focus();
      inp.addEventListener('input', () => {
        document.getElementById('btnSend').classList.toggle('active', inp.value.trim().length > 0);
      });
    }
  }, 50);

  // Scroll
  chat.scrollTop = chat.scrollHeight;
}

function submitAnswer() {
  const inp = document.getElementById('answerInput');
  if (!inp || !inp.value.trim()) return;
  const qs = AI_QUESTIONS[requestType] || AI_QUESTIONS['general'];
  answers[qs[currentQ].field] = inp.value.trim();

  if (currentQ < qs.length - 1) {
    currentQ++;
    renderChat();
  } else {
    showConfirm();
  }
}

function showConfirm() {
  const rt = REQUEST_TYPES[requestType];
  document.getElementById('confirmIcon').textContent = rt.icon;

  let html = `<div class="summary-top">
    <span style="font-size:15px;font-weight:600;font-family:system-ui">${rt.label}</span>
    <span style="font-size:13px;color:var(--sage-700);font-family:system-ui;background:var(--sage-100);padding:4px 12px;border-radius:8px">Est. ${rt.est}</span>
  </div>`;

  if (files.length) {
    html += `<div class="summary-label">Files</div>`;
    files.forEach(f => { html += `<div class="summary-value">📄 ${esc(f.name)}</div>`; });
  }

  if (instruction) {
    html += `<div class="summary-label">Instruction</div><div class="summary-value">${esc(instruction)}</div>`;
  }

  Object.entries(answers).forEach(([k,v]) => {
    html += `<div class="summary-label">${k.replace(/_/g,' ')}</div><div class="summary-value">${esc(v)}</div>`;
  });

  document.getElementById('summaryCard').innerHTML = html;
  showPhase('phaseConfirm');
}

function commitRequest() {
  const id = 'REQ-' + Date.now().toString(36).toUpperCase();
  const rt = REQUEST_TYPES[requestType];

  document.getElementById('requestIdDisplay').textContent = id;
  document.getElementById('trackingDetail').textContent = `${rt.label} — Estimated completion: ${rt.est}`;

  trackingStage = 0;
  renderTracker();
  showPhase('phaseTracking');

  // Show confirmation toast
  showToast('Instruction Committed', `Your request ${id} is now being processed.`);

  // Simulate progression
  const timings = [0, 2200, 4500, 7000, 10000, 13500, 17000];
  const stageToasts = [
    null,
    ['Reading Documents', 'AI is extracting data from your uploaded files...'],
    ['Building Calculations', 'Generating deterministic scripts for your fund rules...'],
    ['Validating', 'Running double-entry, LPA limits, and ACID guardrails...'],
    ['Defence Checks', 'Dual-builder consensus, Z3 formal verification, property testing...'],
    ['Human Review', 'Your fund accountant has been notified for final review.'],
    ['Processing Complete', 'Your results are ready — click View Results to download.'],
  ];

  timings.forEach((t, i) => {
    if (i === 0) return;
    setTimeout(() => {
      trackingStage = i;
      renderTracker();
      if (stageToasts[i]) showToast(stageToasts[i][0], stageToasts[i][1]);
    }, t);
  });
}

function renderTracker() {
  const area = document.getElementById('trackerArea');
  let html = '<div class="tracker-line"></div>';
  const pct = Math.min(100, (trackingStage / (TRACKING_STAGES.length - 1)) * 100);
  html += `<div class="tracker-progress" style="height:${pct}%"></div>`;

  TRACKING_STAGES.forEach((s, i) => {
    const isComplete = i < trackingStage;
    const isActive = i === trackingStage;
    const isPending = i > trackingStage;
    const cls = isPending ? 'track-step pending' : 'track-step';
    const nodeCls = isComplete ? 'complete' : isActive ? 'active' : 'waiting';

    html += `<div class="${cls}">
      <div class="track-node ${nodeCls}">
        ${isComplete ? '<span class="check">✓</span>' : isActive ? '<div class="pulse-dot"></div>' : ''}
      </div>
      <div>
        <div style="display:flex;align-items:center;flex-wrap:wrap">
          <span style="font-size:16px;margin-right:6px">${s.icon}</span>
          <span class="track-label${isActive?' active-label':''}">${s.label}</span>
          ${isActive ? '<span class="track-badge">IN PROGRESS</span>' : ''}
          ${isComplete ? '<span class="track-badge-done">✓ done</span>' : ''}
        </div>
        <div class="track-detail">${s.detail}</div>
      </div>
    </div>`;
  });

  area.innerHTML = html;

  // Show complete card
  const cc = document.getElementById('completeCard');
  if (trackingStage >= TRACKING_STAGES.length - 1) {
    cc.classList.remove('hidden');
    cc.innerHTML = `<div class="complete-card">
      <div class="complete-icon">🎉</div>
      <div class="complete-title">Processing Complete</div>
      <div class="complete-sub">Your results are ready for review</div>
      <button class="btn-results">View Results →</button>
    </div>`;
  } else {
    cc.classList.add('hidden');
  }
}

// ═══════════════════════════════════════════════════════════════
// TOAST NOTIFICATIONS
// ═══════════════════════════════════════════════════════════════
function showToast(title, msg) {
  const container = document.getElementById('toastContainer');
  const toast = document.createElement('div');
  toast.className = 'toast';
  toast.innerHTML = `<div class="toast-title">${esc(title)}</div><div class="toast-msg">${esc(msg)}</div>`;
  container.appendChild(toast);
  setTimeout(() => { toast.classList.add('out'); }, 3500);
  setTimeout(() => { toast.remove(); }, 4000);
}
